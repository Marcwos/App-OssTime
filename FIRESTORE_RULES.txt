rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Regla para la colección 'users'
    match /users/{userId} {
      // Permitir lectura si el usuario está autenticado
      // Esto incluye consultas con whereEqualTo y consultas compuestas
      allow read: if request.auth != null;
      
      // Permitir escritura solo si:
      // 1. El usuario está autenticado Y
      // 2. Está escribiendo su propio documento O es ADMIN
      allow create: if request.auth != null 
                    && (request.auth.uid == userId 
                        || (exists(/databases/$(database)/documents/users/$(request.auth.uid)) 
                            && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'ADMIN'));
      
      allow update: if request.auth != null 
                    && (request.auth.uid == userId 
                        || (exists(/databases/$(database)/documents/users/$(request.auth.uid)) 
                            && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'ADMIN'));
      
      allow delete: if request.auth != null 
                    && exists(/databases/$(database)/documents/users/$(request.auth.uid))
                    && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'ADMIN';
    }

    // Regla para la colección 'schedules'
    match /schedules/{scheduleId} {
      allow read, write: if request.auth != null;
    }

    // Regla para la colección 'classes'
    match /classes/{classId} {
      allow read, write: if request.auth != null;
    }

    // Regla para la colección 'students'
    match /students/{studentId} {
      allow read, write: if request.auth != null;
    }

    // Regla para la colección 'attendances'
    match /attendances/{attendanceId} {
      allow read, write: if request.auth != null;
    }

    // Regla para la colección 'tournaments'
    match /tournaments/{tournamentId} {
      allow read, write: if request.auth != null;
    }
  }
}
